function double(delay,value, suc, fail){
   setTimeout(() =>{
     try {
       if(typeof value != 'number'){
         throw 'value must be a number'
       }
       suc(value*2)
     } catch (e){
       fail(e)
     }
   }
   , delay)
}

const sucCB = (v) => console.log(v)
const failCB = (e) => console.log(e)
double(1000, 3, sucCB, failCB)
double(1000, '3', sucCB, failCB)
#+END_SRC
