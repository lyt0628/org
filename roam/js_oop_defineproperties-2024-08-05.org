let book = {}

Object.defineProperties(book, {
    year_: {
      value: 2017
    },
    edition: {
      value: 1
    },
    year:{
      get() {
        return this.year_ // convention private prop
      },
      set(newValue) {
        if (newValue > 2014) {
          this.year_ = newValue
	  this.edition += newValue - 2017
        }
      }
    }
  }
)

book.year = 2018
console.log(book.edition)
