#+BEGIN_SRC js :noweb yes :results output
propKey = 'name'

cnt = 0
function getUniKey(key){
  return `${key}_${cnt++}`
}

let obj = {
  [getUniKey(propKey)] : 'Matt',
  [getUniKey(propKey)] : 'Koj',
  [getUniKey(propKey)] : 'Falisc'
}

console.log(obj.name_0)
console.log(obj.name_1)
console.log(obj.name_2)
#+END_SRC

: Koj
